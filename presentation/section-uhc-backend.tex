\begin{frame}[fragile]{UHC Backend - Challenges}
\begin{itemize}
\item Agda is a moving target
\item UHC Core was not intended as public API
\item Undocumented assumptions inside UHC
\pause
\begin{lstlisting}[language=Haskell]
case x of
  []     -> a
  (x:xs) -> b
\end{lstlisting}
is not the same as
\begin{lstlisting}[language=Haskell]
case x of
  (x:xs) -> b
  []     -> a
\end{lstlisting}
